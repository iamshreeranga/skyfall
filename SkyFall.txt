void renderBitmapString(float x,float y,float z,void *font,char *string)
{
char *c;
glRasterPos3f(x,y,z);
for(c=string; *c != '\0'; c++)
	glutBitmapCharacter(font, *c);
}

void renderStrokeString(float x,float y,float z,void *font,char *string)
{
char *c;
glTranslatef(x,y,z);
glScalef(0.5,0.5,0);
for(c=string; *c != '\0'; c++)
	glutStrokeCharacter(font, *c);
}

void fall(int c)
{	
	
{
	if(ey1>15)       //speed ge
	{
		
	ey1=ey1-4;
	
	}
	else
	{
		if(temp1>=bx1 && temp1<=bx1+80)  //If ball is caught by below bar
		{
		time(&seconds);
	 srand((unsigned int) seconds);
	 
		temp1=a[rand()%5];
		ey1=700;
		score=score+10;
			
		}
		else                              //couldn't catch
		goflag=1;

	}
}

	glutTimerFunc(10,fall,c+50);
		glutPostRedisplay();
}


void mykey(int key,int x,int y)
{
	switch(key)
	{
	
	case GLUT_KEY_RIGHT:if(basva==0)
		if(bx1<700)
		{
			bx1=bx1+20;
		}
		
			break;
	case GLUT_KEY_LEFT:if(basva==0)
			if(bx1>0)
			{
				bx1=bx1-20;
			}
			
			break;

case GLUT_KEY_UP:
		ey1=700;
		goflag=0;
			wflag=0;
			
	}
	glutPostRedisplay();
}
void myinit()
{
	
		glMatrixMode(GL_PROJECTION);
		glShadeModel (GL_SMOOTH);
		glLoadIdentity();
		glOrtho(0,700,0,700,-20,20);
		glEnable(GL_DEPTH_TEST);
		glMatrixMode(GL_MODELVIEW);
}

void mymenu(int ch)
{

	{
	case 1: exit(0);
		break;
	}
	glutPostRedisplay();

}

void main(int argc, char ** argv)
{
